def preprocess(data, webstat, web_flag=False):
    data['utm_source'] = data['utm_source'].fillna(-1)
    data['utm_source'] = data['utm_source'].astype('int32')
    webstat['product_in_sale'] = webstat['product_in_sale'].fillna(0).astype('int32')
    webstat['price'] = webstat['price'].fillna(0).astype('float')
    webstat['good_id'] = webstat['good_id'].fillna(0).astype('int32')

    data['create_time'] = pd.to_datetime(data['create_time'])
    data['model_create_time'] = pd.to_datetime(data['model_create_time'])
    webstat['date_time'] = pd.to_datetime(webstat['date_time'])
    
    # Для начала давайте подсчитаем количество сессий на каждый order_id
    sessions_count = webstat['sessionkey_id'].value_counts().reset_index()
    sessions_count.columns = ['sessionkey_id', 'sessions_per_order']
    data = pd.merge(data, sessions_count, on='sessionkey_id', how='left')
    data['sessions_per_order'] = data['sessions_per_order'].fillna(0).astype('int32')

    # Кроме того, хочу агрегировать признак pageview_duration_sec, который как мне кажется интересным
    session_durations = webstat.groupby('sessionkey_id')['pageview_duration_sec'].sum().reset_index()
    session_durations.columns = ['sessionkey_id', 'total_duration_per_session']
    data = pd.merge(data, session_durations, on='sessionkey_id', how='left')
    data['total_duration_per_session'] = data['total_duration_per_session'].fillna(0).astype('int32')

    # Кроме суммы давайте попробуем агрегировать pageview_duration_sec как среднее
    mean_session_durations = webstat.groupby('sessionkey_id')['pageview_duration_sec'].mean().reset_index()
    mean_session_durations.columns = ['sessionkey_id', 'mean_duration_per_session']
    data = pd.merge(data, mean_session_durations, on='sessionkey_id', how='left')
    data['mean_duration_per_session'] = data['mean_duration_per_session'].fillna(0).astype('int32')

    # Как узнали ранее давайте запишем в data знание о том, сколько было не пропущенных значений в последних трёх колонках webstat
    nans_field = webstat.groupby('sessionkey_id')['product_in_sale'].sum().reset_index()
    nans_field.columns = ['sessionkey_id', 'nans_field']
    data = pd.merge(data, nans_field, on='sessionkey_id', how='left')
    data['nans_field'] = data['nans_field'].fillna(0).astype('int32')

    # так же попробуем воспользоваться знанием о среднем количестве не нанов на один order_id
    mean_nans = data.nans_field / (data.sessions_per_order + 1)
    data['mean_nans'] = mean_nans

    # агрегация информации о page_type. Давайте попробуем вставить самое часто встречаемое значение
    page_type_counts = webstat.groupby(['sessionkey_id', 'page_type']).size().reset_index(name='counts')
    page_type_counts_sorted = page_type_counts.sort_values(by=['sessionkey_id', 'counts'], ascending=[True, False])
    most_frequent_page_type = page_type_counts_sorted.drop_duplicates(subset=['sessionkey_id'])
    data = pd.merge(data, most_frequent_page_type[['sessionkey_id', 'page_type']], on='sessionkey_id', how='left')
    data['page_type'] = data['page_type'].fillna(-1).astype('int32')

    # Попробуем исследовать разность дат между таблицами train И webstat
    arg_date_web = webstat.groupby('sessionkey_id')['date_time'].mean().reset_index()
    tmp_time = pd.merge(data[['order_id', 'create_time', 'sessionkey_id']], arg_date_web, on='sessionkey_id', how='left')
    tmp_time['diff_time'] = tmp_time['date_time'] - tmp_time['create_time']
    data = pd.merge(data, tmp_time[['sessionkey_id', 'diff_time']], on='sessionkey_id', how='left')
    data['diff_time'] = data['diff_time'].fillna(pd.to_timedelta(0)).dt.days

    data['root_id'] = data['root_id'].clip(upper=2500)

    data['rating_value'] = data['rating_value'].fillna(-1).astype('int32')
    data['rating_count'] = data['rating_count'].fillna(-1).astype('int32')

    data = data.drop(['good_id',
                      'category_id',
                      'model_id', 
                      'model_create_time',
                      'description_length',
                      'goods_qty',], axis=1)
    
    if web_flag:
        return data.copy(), webstat.copy()
    return data.copy()

def lgb_model_train(web_train, web_val):

                        label=web_train['is_callcenter'], free_raw_data=False)

    )

    return model

def web_proc(web, data, cat_cols):
    # Заполняем пропуски
    web['product_in_sale'] = web['product_in_sale'].fillna(0).astype('int32')
    web['price'] = web['price'].fillna(0).astype('float')
    web['good_id'] = web['good_id'].fillna(0).astype('int32')

    # Считаем количество одинаковых сессий 
    tmp = web['sessionkey_id'].value_counts().reset_index()
    tmp.columns = ['sessionkey_id', 'sessions_per_order']
    web = pd.merge(web, tmp, on='sessionkey_id', how='left')
    web['sessions_per_order'] = web['sessions_per_order'].fillna(-1).astype('int32')

    # Кроме того, хочу агрегировать признак pageview_duration_sec, который как мне кажется интересным
    web['pageview_duration_sec'] = web['pageview_duration_sec'].clip(lower=0, upper=200)
    session_durations = web.groupby('sessionkey_id')['pageview_duration_sec'].sum().reset_index()
    session_durations.columns = ['sessionkey_id', 'total_duration_per_session']
    web = pd.merge(web, session_durations, on='sessionkey_id', how='left')
    web['total_duration_per_session'] = web['total_duration_per_session'].fillna(0).astype('int32')

    # Кроме суммы давайте попробуем агрегировать pageview_duration_sec как среднее
    mean_session_durations = web.groupby('sessionkey_id')['pageview_duration_sec'].mean().reset_index()
    mean_session_durations.columns = ['sessionkey_id', 'mean_duration_per_session']
    web = pd.merge(web, mean_session_durations, on='sessionkey_id', how='left')
    web['mean_duration_per_session'] = web['mean_duration_per_session'].fillna(0).astype('int32')

    # Как узнали ранее давайте запишем в web знание о том, сколько было не пропущенных значений в последних трёх колонках webstat
    nans_field = webstat.groupby('sessionkey_id')['product_in_sale'].sum().reset_index()
